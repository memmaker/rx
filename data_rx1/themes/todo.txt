
func (u *UI) setAmberStyle() {
	amberTerminal := tcell.StyleDefault.
		Foreground(tcell.NewRGBColor(255, 162, 1)).
		Background(tcell.NewRGBColor(13, 8, 1))
	u.defaultStyle = amberTerminal
	u.textColor = color.RGBA{255, 162, 1, 255}
	u.hiLiteColor = color.RGBA{255, 180, 51, 255}
}

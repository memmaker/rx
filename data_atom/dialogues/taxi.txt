%rec: OpeningBranch

cond: !HasFlag('Killed(taxi_driver)')
goto: NoTaxi

cond: HasFlag('Killed(taxi_driver)') && !HasFlag('searched_taxi')
goto: SearchTaxi

cond: HasFlag('Killed(taxi_driver)') && HasFlag('searched_taxi')
goto: UseTaxi
